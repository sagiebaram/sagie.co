  reValidateMode: 'onChange',
  shouldFocusError: true,
})

- `fields` useState object — replaced by RHF

- `validate()` function — replaced by zodResolver
- `set()` helper — no longer needed

**Hidden honeypot field:** Keep the existing hidden input. If it currently uses useState, switch to: `<input type="text" style={{ display: 'none' }} onChange={e => { trapRef.current = e.target.value }} tabIndex={-1} autoComplete="off" />`

**onSubmit handler:**
```typescript
const onSubmit = async (data: FormData) => {
  if (rateLimitUntil !== null && Date.now() < rateLimitUntil) return
  setSubmitWarning(null)
  try {
    const res = await fetch('/api/applications/membership', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ ...data, _trap: trapRef.current, _t: loadTime.current }),
    })
    if (res.status === 429) {
      // Keep exact Phase 5 rate limit handling: parse Retry-After, set submitWarning, set rateLimitUntil
    }
    if (!res.ok) throw new Error()
    setSuccess(true)
  } catch {
    // Keep generic error display
  }
}
```
